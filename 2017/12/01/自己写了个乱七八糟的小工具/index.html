自己写了个乱七八糟的小工具

发表于 2017-12-01 | 阅读次数:

Tags : 通过反射给自定义类设置属性生成json格式的数据

###不完善的地方###
对于自定义类中的字段：容器只写了List，再遇到其他再加。数组一般不用了就没写。还有一些字段值会生成null，到时候自己手动加数据吧，比new个对象再挨个set值或者直接拼写json格式数据方便多了。

###用法###
只需要配置你对应类的位置就可以了。使用的阿里的fastjson包。（代码乱七八糟，还未完善，有时间兴趣可以完善下）

代码

public class BeanInstantiation {

    public static void main(String[] args) throws Exception{
        Random rand = new Random(47);
        int nextInt = rand.nextInt(10);
        Long nextLong = rand.nextLong();

        List list = new ArrayList();
        list.add("12");
        list.add("12123");

        Class<Object> clazz =  (Class<Object>) Class.forName("填写包的位置（当然是可以访问到的位置）");

        Object obj = m(clazz, nextInt, nextLong, list);
        String jsonString = JSONObject.toJSONString(obj, true);
        System.out.println(jsonString);
    }



    /**
    * 返回null的要处理一下，包装类的问题。包装类的处理在泛型那里也可以处理。
    * @param clazz
    * @param nextInt
    * @param nextLong
    * @param list
    * @return
    * @throws Exception
    */
    public static Object m(Class clazz, int nextInt, Long nextLong,List list) throws Exception{

        if(clazz == null){
            return null;
        }
        //不能传入包装类
        if(clazz.getName() == "java.lang.Integer"){
            return null;
        }
        if(clazz.getName() == "java.lang.Long"){
            return null;
        }


        Object obj = clazz.newInstance();


        if(obj.equals("")){
            System.out.println("=======inner中list有泛型=======");
            return null;
        }
        Field[] files = clazz.getDeclaredFields();
        for(Field file: files){
            file.setAccessible(true);
            String typeName = file.getType().getName();
            String filename = file.getName();

            if(typeName.equals("java.lang.String")){
                //file.set
                file.set(obj, "这是一个"+filename);
            }
            else if(typeName.equals("int")){
                file.set(obj, nextInt);
            }
            else if(typeName.equals("java.lang.Integer")){
                file.set(obj, nextInt);
            }
            else if(typeName.equals("java.lang.Long")){
                file.set(obj, nextLong);
            }
            //是否可以将泛型和myname中的自定义对象一起在下面这个if中实现
            //获得list中的对象
            else if(typeName.equals("java.util.List")){

                Type genericType = file.getGenericType();

                System.out.println("genericType:"+genericType);
                if(genericType == null){
                    continue;
                }
                if(genericType instanceof ParameterizedType){
                    ParameterizedType pt = (ParameterizedType)genericType;
                    //class com.liukai.LK
                    Class<?> genericClazz = (Class<?>)pt.getActualTypeArguments()[0];

                    String genericName = genericClazz.getName();
                    System.out.println("genericClazz:" + genericClazz);
                    System.out.println("genericName:" + genericName);
                    //对泛型反射
                    Class clazz2 = Class.forName(genericName);


                    //递归，给对象set
                    Object m = m(clazz2, nextInt,nextLong,list);

                    /*Object setvalue = setvalue(clazz2);*/
                    List<Object> arrayList = new ArrayList<>();
                    arrayList.add(m);

                    file.set(obj, arrayList);
                }else{
                    file.set(obj, list);
                }

            }
            //处理自定义对象
            else{
                System.out.println("-------------"+typeName);
                System.out.println("-------------"+typeName);
                Class clazz2 = Class.forName(typeName);
                Object m = m(clazz2, nextInt,nextLong,list);
                file.set(obj, m);
            }


            System.out.println(typeName);

        }

        return obj;
    }
}
